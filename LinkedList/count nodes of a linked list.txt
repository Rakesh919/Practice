class Solution
{
    //Function to count nodes of a linked list.
    public static int getCount(Node head)
    {
        
        //Code here
        int count=0;
        Node cur =head;
        while(cur!=null){
            cur =cur.next;
            count++;
        }
    
        return count;
    }
    
}
