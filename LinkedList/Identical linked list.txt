    public boolean isIdentical (Node head1, Node head2){

        int a =(findLength(head1));

        while(cur!=null && pre!=null){
            
                if(cur.data !=pre.data){
                return false;
                
               
                
            }
             cur =cur.next;
                pre =pre.next;
        }

    }
    
    int findLength(Node head){
        Node cur =head;
        int i=0;
        while(cur!=null){
            i++;
            cur=cur.next;
        }
        return i;
    }
