/* linked list node class:

class Node {
    int value;
    Node next;
    Node(int value) {
        this.value = value;
    }
}

*/

class Solution
{
    //Function to reverse a linked list.
    Node reverseList(Node head)
    {
        // code here
        if(head.next == null){
            return head;
        }
        
        Node cur = head;
        Node prev =null;
        
        while(cur!=null){
            Node temp =cur.next;
            cur.next =prev;
            prev =cur;
            cur =temp;
        }
        return prev;
    }
}
